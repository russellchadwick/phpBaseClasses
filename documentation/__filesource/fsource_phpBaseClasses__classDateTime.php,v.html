  245.                 list ($year, $month, $day) = explode ('-', $datepart);

  345.         /**
  346.          * Returns date and time data for a given timestamp
  347.          *
  348.          * @access private
  349.          * @param integer Timestamp since January 1st of beginYear
  350.          * @return array Date and time data (year, month, day, hour, minute, second, dayofweek, dayofyear)
  351.          */
  352.         function getTimeDataFromTimestamp ($timestamp) {
  353.             $year = $this->_beginYear;
  354.  
  355.             $totaldays = 0;
  356.  
  357.             while ($timestamp > $this->getSecondsPerYear ($year)) {
  358.                 $timestamp -= $this->getSecondsPerYear ($year);
  359.                 $totaldays += $this->getDaysPerYear ($year);
  360.                 $year++;
  361.             }
  362.  
  363.             $month = 1;
  364.             $dayofyear = 0;
  365.  
  366.             while (($timestamp >= $this->getSecondsPerMonth ($year, $month)) && ($month < 13)) {
  367.                 $timestamp -= $this->getSecondsPerMonth ($year, $month);
  368.                 $dayofyear += $this->getDaysPerMonth ($year, $month);
  369.                 $month++;
  370.             }
  371.  
  372.             if ($month == 13) {
  373.                 $year++;
  374.                 $month -= 12;
  375.             }
  376.  
  377.             $day = floor ($timestamp / $this->_secondsPerDay) + 1;
  378.             $dayofyear += $day - 1;
  379.             $totaldays += $dayofyear;
  380.             $hour = floor (($timestamp % $this->_secondsPerDay) / (3600));
  381.             $minute = floor ((($timestamp % $this->_secondsPerDay) % 3600) / 60);
  382.             $second = floor ((($timestamp % $this->_secondsPerDay) % 3600) % 60);
  383.  
  384.             $dayofweek = (($this->_beginDayofWeek + $totaldays) % 7);
  385.  
  386.             return array ($year, $month, $day, $hour, $minute, $second, $dayofweek, $dayofyear);
  387.         }

  404.         /**

  406.          * Leaving format empty will unlock
  407.          *
  408.          * @access public

  410.          * @return boolean Whether successful, fails when date format is not found
  411.          */
  412.         function lockDateFormat ($format='') {
  413.             if (empty($format)) {
  414.                 $this->_lockFormat = '';
  415.                 return true;
  416.             } else {
  417.                 if (isset($this->_knownFormats[$format])) {
  418.                     $this->_lockFormat = $format;
  419.                     return true;
  420.                 } else
  421.                     return false;
  422.             }
  423.         }

  425.         /**
  426.          * Returns the timestamp since January 1st of beginYear in UTC
  427.          * Behaves exactly like PHP's implementation of unix mktime
  428.          * http://www.php.net/mktime
  429.          *
  430.          * @access public
  431.          * @param integer Empty will cause lookup of current hour
  432.          * @param integer Empty will cause lookup of current minute
  433.          * @param integer Empty will cause lookup of current second
  434.          * @param integer Empty will cause lookup of current month
  435.          * @param integer Empty will cause lookup of current day
  436.          * @param integer Empty will cause lookup of current year
  437.          * @return integer Timestamp since January 1st of beginYear in UTC
  438.          */
  439.         function mktime ($hour='', $minute='', $second='', $month='', $day='', $year='') {
  440.             debug ('DateTime::mktime (' . $hour . ', ' . $minute . ', ' . $second . ', ' . $month . ', ' . $day . ', ' . $year . ')', 4);
  441.  
  442.             $timestamp = 0;
  443.  
  444.             if ($year === '') 
  445.                 $year = date ('Y');
  446.  
  447.             if ($year > $this->_beginYear)
  448.                 foreach (range ($this->_beginYear, ($year - 1)) as $curyear) 
  449.                     $timestamp += $this->getSecondsPerYear ($curyear);
  450.  
  451.             if ($month === '') 
  452.                 $month = date ('m');
  453.  
  454.             foreach (range (0, ($month - 1)) as $curmonth) 
  455.                 $timestamp += $this->getSecondsPerMonth ($year, $curmonth);
  456.  
  457.             if ($day === '')
  458.                 $day = date ('d');
  459.  
  460.             $timestamp += (($day - 1) * $this->_secondsPerDay);
  461.  
  462.             if ($hour === '')
  463.                 $hour = date ('H');
  464.  
  465.             $timestamp += ($hour * 3600);
  466.  
  467.             if ($minute === '') 
  468.                 $minute = date ('i');
  469.  
  470.             $timestamp += ($minute * 60);
  471.  
  472.             if ($second === '')
  473.                 $second = date ('s');
  474.  
  475.             $timestamp += $second;
  476.  
  477.             if ($this->_timezoneAdjust) {
  478.                 $timezone = date ('Z', mktime ($hour, $minute, $second, $month, $day, 2000));
  479.                 $timestamp -= $timezone;
  480.             }
  481.  
  482.             return $timestamp;
  483.         }
  484.  
  485.         /**
  486.          * Returns a formatted date
  487.          * Behaves exactly like PHP's implementation of unix date
  488.          * http://www.php.net/date
  489.          * 2 new parameters added
  490.          * X: # of weekday for this month
  491.          * x: Total # of weekdays for this month
  492.          *
  493.          * @access public
  494.          * @param string Format to return date in
  495.          * @param integer Timestamp since January 1st of beginYear in UTC
  496.          * @return string Formatted date
  497.          */
  498.         function date ($format, $timestamp=0) {
  499.             if (empty($timestamp)) 
  500.                 $timestamp = $this->mktime ();
  501.  
  502.             $timestamp = $this->convertToTimestamp ($timestamp, false);
  503.  
  504.             list ($year, $month, $day, $hour, $minute, $second, $dayofweek, $dayofyear) = $this->getTimeDataFromTimestamp ($timestamp);
  505.  
  506.             if ($this->_timezoneAdjust) {
  507.                 $timezone_info = explode (' ', date ('O T Z I', mktime ($hour, $minute, $second, $month, $day, 2000)));
  508.                 $timestamp += $timezone_info[2];
  509.  
  510.                 list ($year, $month, $day, $hour, $minute, $second, $dayofweek, $dayofyear) = $this->getTimeDataFromTimestamp ($timestamp);
  511.             }
  512.  
  513.             $retval = '';
  514.  
  515.             $format = str_replace ('r', 'D, d M Y H:i:s O', $format);
  516.  
  517.             for ($index = 0; $index < strlen ($format); $index++) {
  518.                 switch (substr ($format, $index, 1)) {
  519.                     case 'a':
  520.                         if ($hour > 12)
  521.                             $retval .= 'pm';
  522.                         else
  523.                             $retval .= 'am';
  524.                         break;
  525.                     case 'A':
  526.                         if ($hour > 12)
  527.                             $retval .= 'PM';
  528.                         else
  529.                             $retval .= 'AM';
  530.                         break;
  531.                     case 'D':
  532.                         $retval .= substr ($this->_dayOfWeekNames[$dayofweek], 0, 3);
  533.                         break;
  534.                     case 'd':
  535.                         $retval .= str_pad ($day, 2, '0', STR_PAD_LEFT);
  536.                         break;
  537.                     case 'F':
  538.                         $retval .= $this->_monthNames[$month];
  539.                         break;
  540.                     case 'g':
  541.                         if ($hour > 12) 
  542.                             $retval .= $hour - 12;
  543.                         else 
  544.                             $retval .= $hour;
  545.                         break;
  546.                     case 'G':
  547.                         $retval .= $hour;
  548.                         break;
  549.                     case 'h':
  550.                         if ($hour > 12) 
  551.                             $retval .= str_pad ($hour - 12, 2, '0', STR_PAD_LEFT);
  552.                         else 
  553.                             $retval .= str_pad ($hour, 2, '0', STR_PAD_LEFT);
  554.                         break;
  555.                     case 'H':
  556.                         $retval .= str_pad ($hour, 2, '0', STR_PAD_LEFT);
  557.                         break;
  558.                     case 'I':
  559.                         $retval .= $timezone_info[3];
  560.                         break;
  561.                     case 'i':
  562.                         $retval .= str_pad ($minute, 2, '0', STR_PAD_LEFT);
  563.                         break;
  564.                     case 'j':
  565.                         $retval .= $day;
  566.                         break;
  567.                     case 'L':
  568.                         $retval .= (int) $this->isLeapYear ($year);
  569.                         break;
  570.                     case 'l':
  571.                         $retval .= $this->_dayOfWeekNames[$dayofweek];
  572.                         break;
  573.                     case 'M':
  574.                         $retval .= substr ($this->_monthNames[$month], 0, 3);
  575.                         break;
  576.                     case 'm':
  577.                         $retval .= str_pad ($month, 2, '0', STR_PAD_LEFT);
  578.                         break;
  579.                     case 'n':
  580.                         $retval .= $month;
  581.                         break;
  582.                     case 'O':
  583.                         $retval .= $timezone_info[0];
  584.                         break;
  585.                     case 'S':
  586.                         if ($day == 1) 
  587.                             $retval .= 'st';
  588.                         else if ($day == 2) 
  589.                             $retval .= 'nd';
  590.                         else if ($day == 3)
  591.                             $retval .= 'rd';
  592.                         else 
  593.                             $retval .= 'th';
  594.                         break;
  595.                     case 's':
  596.                         $retval .= str_pad ($second, 2, '0', STR_PAD_LEFT);
  597.                         break;
  598.                     case 'T':
  599.                         $retval .= $timezone_info[1];
  600.                         break;
  601.                     case 't':
  602.                         $retval .= $this->getDaysPerMonth ($year, $month);
  603.                         break;
  604.                     case 'U':
  605.                         $retval .= $timestamp;
  606.                         break;
  607.                     case 'W':
  608.                         $retval .= floor ($dayofyear / 7);
  609.                         break;
  610.                     case 'w':
  611.                         $retval .= $dayofweek;
  612.                         break;
  613.                     case 'X': // # of weekday for this month
  614.                         $retval .= floor (($day - 1) / 7) + 1;
  615.                         break;
  616.                     case 'x': // Total # of weekdays for this month
  617.                         $lastdayofweek = ((7 + $dayofweek + (($this->getDaysPerMonth ($year, $month) - $day) % 7)) % 7);
  618.                         if ((($dayofweek + 7) >= ($lastdayofweek - (($this->getDaysPerMonth ($year, $month) % 7) - 1)) + 7) && (($dayofweek + 7) <= ($lastdayofweek + 7))) {
  619.                             $retval .= 5;
  620.                         } else {
  621.                             $retval .= 4;
  622.                         }
  623.                         break;
  624.                     case 'Y':
  625.                         $retval .= $year;
  626.                         break;
  627.                     case 'y':
  628.                         $retval .= substr ($year, 2);
  629.                         break;
  630.                     case 'Z':
  631.                         $retval .= $timezone_info[2];
  632.                         break;
  633.                     case 'z':
  634.                         $retval .= $dayofyear;
  635.                         break;
  636.                     default:
  637.                         $retval .= substr ($format, $index, 1);
  638.                 }
  639.             }
  640.  
  641.             return $retval;
  642.         }

  653.         function getdate ($timestamp=0) {

  666.         /**
  667.          * Returns the number of days between 2 dates regardless of their formats
  668.          *
  669.          * @access public
  670.          * @param string First Date
  671.          * @param string Second Date
  672.          * @param boolean Whether to only count business days
  673.          * @param boolean Whether to always return positive number or not
  674.          * @return integer Number of days between first date and second date
  675.          */
  676.         function daysBetween ($date1='', $date2='', $business_days=false, $absolute_value=true) {
  677.             if (empty($date1)) 
  678.                 $date1 = $this->mktime ();
  679.  
  680.             if (empty($date2)) 
  681.                 $date2 = $this->mktime ();
  682.  
  683.             if ($business_days) {
  684.                 $date1 = $this->getFormattedDate ($date1, $format='ISO8601');
  685.                 $date2 = $this->getFormattedDate ($date2, $format='ISO8601');
  686.  
  687.                 $added_days = 0;
  688.                 if ($date1 == $date2) {
  689.                     return 0;
  690.                 } else if ($date1 > $date2) {
  691.                     do {
  692.                         $added_days++;
  693.                         $date2 = $this->addDays ($date2, 1, true);
  694.                     } while ($date1 > $date2);
  695.  
  696.                     return $added_days;
  697.                 } else {
  698.                     do {
  699.                         $added_days++;
  700.                         $date1 = $this->addDays ($date1, 1, true);
  701.                     } while ($date2 > $date1);
  702.  
  703.                     if ($absolute_value) {
  704.                         return $added_days;
  705.                     } else {
  706.                         return (0 - $added_days);
  707.                     }
  708.                 }
  709.             } else {
  710.                 $date1 = $this->convertToTimestamp ($date1);
  711.                 $date2 = $this->convertToTimestamp ($date2);
  712.  
  713.                 if ($absolute_value) {
  714.                     $difference = abs ($date1 - $date2);
  715.                 } else {
  716.                     $difference = $date1 - $date2;
  717.                 }
  718.  
  719.                 return ceil ($difference / $this->_secondsPerDay);
  720.             }
  721.         }

  825.         /**
  826.          * Compares 2 dates regardless of their format and returns 0 if 2 dates are equal, 1 if second date is greater than first date, and otherwise -1
  827.          *
  828.          * @access public
  829.          * @param string First Date
  830.          * @param string Second Date
  831.          * @return integer 0 if 2 dates are equal, 1 if second date is greater than first date, and otherwise -1
  832.          */
  833.         function dateCompare ($date1='', $date2='') {
  834.             if (empty($date1)) 
  835.                 $date1 = $this->mktime ();
  836.  
  837.             if (empty($date2)) 
  838.                 $date2 = $this->mktime ();
  839.  
  840.             $date1 = $this->convertToTimestamp ($date1);
