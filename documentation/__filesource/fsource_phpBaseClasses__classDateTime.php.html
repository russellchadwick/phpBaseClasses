  214.             } else if (($hypen_count == 2) && ($colon_count == 0)) {

  216.             } else if (($slash_count == 2) && ($colon_count == 2)) {

  221.                 $date_format = 'German with Time';
  222.             } else if (($point_count == 2) && ($colon_count == 0)) {

  475.         /**
  476.          * Returns date and time data for a given timestamp
  477.          *
  478.          * @access private
  479.          * @param integer Timestamp since January 1st of beginYear
  480.          * @return array Date and time data (year, month, day, hour, minute, second, dayofweek, dayofyear)
  481.          */
  482.         function getTimeDataFromTimestamp ($timestamp) {
  483.             $year = $this->_beginYear;
  484.  
  485.             $totaldays = 0;
  486.  
  487.             while ($timestamp > $this->getSecondsPerYear ($year)) {
  488.                 $timestamp -= $this->getSecondsPerYear ($year);
  489.                 $totaldays += $this->getDaysPerYear ($year);
  490.                 $year++;
  491.             }
  492.  
  493.             $month = 1;
  494.             $dayofyear = 0;
  495.  
  496.             while (($timestamp >= $this->getSecondsPerMonth ($year, $month)) && ($month < 13)) {
  497.                 $timestamp -= $this->getSecondsPerMonth ($year, $month);
  498.                 $dayofyear += $this->getDaysPerMonth ($year, $month);
  499.                 $month++;
  500.             }
  501.  
  502.             if ($month == 13) {
  503.                 $year++;
  504.                 $month -= 12;
  505.             }
  506.  
  507.             $day = floor ($timestamp / $this->_secondsPerDay) + 1;
  508.             $dayofyear += $day - 1;
  509.             $totaldays += $dayofyear;
  510.             $hour = floor (($timestamp % $this->_secondsPerDay) / (3600));
  511.             $minute = floor ((($timestamp % $this->_secondsPerDay) % 3600) / 60);
  512.             $second = floor ((($timestamp % $this->_secondsPerDay) % 3600) % 60);
  513.  
  514.             $dayofweek = (($this->_beginDayofWeek + $totaldays) % 7);
  515.  
  516.             return array ($year, $month, $day, $hour, $minute, $second, $dayofweek, $dayofyear);
  517.         }

  557.         /**
  558.          * Returns the timestamp since January 1st of beginYear in UTC
  559.          * Behaves exactly like PHP's implementation of unix mktime
  560.          * http://www.php.net/mktime
  561.          *
  562.          * @access public
  563.          * @param integer Empty will cause lookup of current hour
  564.          * @param integer Empty will cause lookup of current minute
  565.          * @param integer Empty will cause lookup of current second
  566.          * @param integer Empty will cause lookup of current month
  567.          * @param integer Empty will cause lookup of current day
  568.          * @param integer Empty will cause lookup of current year
  569.          * @return integer Timestamp since January 1st of beginYear in UTC
  570.          */
  571.         function mktime ($hour='', $minute='', $second='', $month='', $day='', $year='') {
  572.             debug ('DateTime::mktime (' . $hour . ', ' . $minute . ', ' . $second . ', ' . $month . ', ' . $day . ', ' . $year . ')', 4);
  573.  
  574.             $timestamp = 0;
  575.  
  576.             if ($year === '') {
  577.                 $year = date ('Y');
  578.             }
  579.  
  580.             if ($year > $this->_beginYear) {
  581.                 foreach (range ($this->_beginYear, ($year - 1)) as $curyear) {
  582.                     $timestamp += $this->getSecondsPerYear ($curyear);
  583.                 }
  584.             }
  585.  
  586.             if ($month === '') {
  587.                 $month = date ('m');
  588.             }
  589.  
  590.             foreach (range (0, ($month - 1)) as $curmonth) {
  591.                 $timestamp += $this->getSecondsPerMonth ($year, $curmonth);
  592.             }
  593.  
  594.             if ($day === '') {
  595.                 $day = date ('d');
  596.             }
  597.  
  598.             $timestamp += (($day - 1) * $this->_secondsPerDay);
  599.  
  600.             if ($hour === '') {
  601.                 $hour = date ('H');
  602.             }
  603.  
  604.             $timestamp += ($hour * 3600);
  605.  
  606.             if ($minute === '') {
  607.                 $minute = date ('i');
  608.             }
  609.  
  610.             $timestamp += ($minute * 60);
  611.  
  612.             if ($second === '') {
  613.                 $second = date ('s');
  614.             }
  615.  
  616.             $timestamp += $second;
  617.  
  618.             if ($this->_timezoneAdjust) {
  619.                 $timezone = date ('Z', mktime ($hour, $minute, $second, $month, $day, 2000));
  620.                 $timestamp -= $timezone;
  621.             }
  622.  
  623.             return $timestamp;
  624.         }
  625.  
  626.         /**
  627.          * Returns a formatted date
  628.          * Behaves exactly like PHP's implementation of unix date
  629.          * http://www.php.net/date
  630.          * 2 new parameters added
  631.          * X: # of weekday for this month
  632.          * x: Total # of weekdays for this month
  633.          *
  634.          * @access public
  635.          * @param string Format to return date in
  636.          * @param integer Timestamp since January 1st of beginYear in UTC
  637.          * @return string Formatted date
  638.          */
  639.         function date ($format, $timestamp=0) {
  640.             debug ('DateTime::date (' . $format . ', ' . $timestamp . ')', 4);
  641.  
  642.             if (empty ($timestamp)) {
  643.                 $timestamp = $this->mktime ();
  644.             }
  645.  
  646.             $timestamp = $this->convertToTimestamp ($timestamp, false);
  647.  
  648.             list ($year, $month, $day, $hour, $minute, $second, $dayofweek, $dayofyear) = $this->getTimeDataFromTimestamp ($timestamp);
  649.  
  650.             if ($this->_timezoneAdjust) {
  651.                 $timezone_info = explode (' ', date ('O T Z I', mktime ($hour, $minute, $second, $month, $day, 2000)));
  652.                 $timestamp += $timezone_info[2];
  653.  
  654.                 list ($year, $month, $day, $hour, $minute, $second, $dayofweek, $dayofyear) = $this->getTimeDataFromTimestamp ($timestamp);
  655.             }
  656.  
  657.             $retval = '';
  658.  
  659.             $format = str_replace ('r', 'D, d M Y H:i:s O', $format);
  660.  
  661.             for ($index = 0; $index < strlen ($format); $index++) {
  662.                 switch (substr ($format, $index, 1)) {
  663.                     case 'a':
  664.                         if ($hour > 12) {
  665.                             $retval .= 'pm';
  666.                         } else {
  667.                             $retval .= 'am';
  668.                         }
  669.                         break;
  670.                     case 'A':
  671.                         if ($hour > 12) {
  672.                             $retval .= 'PM';
  673.                         } else {
  674.                             $retval .= 'AM';
  675.                         }
  676.                         break;
  677.                     case 'D':
  678.                         $retval .= substr ($this->_dayOfWeekNames[$dayofweek], 0, 3);
  679.                         break;
  680.                     case 'd':
  681.                         $retval .= str_pad ($day, 2, '0', STR_PAD_LEFT);
  682.                         break;
  683.                     case 'F':
  684.                         $retval .= $this->_monthNames[$month];
  685.                         break;
  686.                     case 'g':
  687.                         if ($hour > 12) {
  688.                             $retval .= $hour - 12;
  689.                         } else {
  690.                             $retval .= $hour;
  691.                         }
  692.                         break;
  693.                     case 'G':
  694.                         $retval .= $hour;
  695.                         break;
  696.                     case 'h':
  697.                         if ($hour > 12) {
  698.                             $retval .= str_pad ($hour - 12, 2, '0', STR_PAD_LEFT);
  699.                         } else {
  700.                             $retval .= str_pad ($hour, 2, '0', STR_PAD_LEFT);
  701.                         }
  702.                         break;
  703.                     case 'H':
  704.                         $retval .= str_pad ($hour, 2, '0', STR_PAD_LEFT);
  705.                         break;
  706.                     case 'I':
  707.                         $retval .= $timezone_info[3];
  708.                         break;
  709.                     case 'i':
  710.                         $retval .= str_pad ($minute, 2, '0', STR_PAD_LEFT);
  711.                         break;
  712.                     case 'j':
  713.                         $retval .= $day;
  714.                         break;
  715.                     case 'L':
  716.                         $retval .= (int) $this->isLeapYear ($year);
  717.                         break;
  718.                     case 'l':
  719.                         $retval .= $this->_dayOfWeekNames[$dayofweek];
  720.                         break;
  721.                     case 'M':
  722.                         $retval .= substr ($this->_monthNames[$month], 0, 3);
  723.                         break;
  724.                     case 'm':
  725.                         $retval .= str_pad ($month, 2, '0', STR_PAD_LEFT);
  726.                         break;
  727.                     case 'n':
  728.                         $retval .= $month;
  729.                         break;
  730.                     case 'O':
  731.                         $retval .= $timezone_info[0];
  732.                         break;
  733.                     case 'S':
  734.                         if ($day == 1) {
  735.                             $retval .= 'st';
  736.                         } else if ($day == 2) {
  737.                             $retval .= 'nd';
  738.                         } else if ($day == 3) {
  739.                             $retval .= 'rd';
  740.                         } else {
  741.                             $retval .= 'th';
  742.                         }
  743.                         break;
  744.                     case 's':
  745.                         $retval .= str_pad ($second, 2, '0', STR_PAD_LEFT);
  746.                         break;
  747.                     case 'T':
  748.                         $retval .= $timezone_info[1];
  749.                         break;
  750.                     case 't':
  751.                         $retval .= $this->getDaysPerMonth ($year, $month);
  752.                         break;
  753.                     case 'U':
  754.                         $retval .= $timestamp;
  755.                         break;
  756.                     case 'W':
  757.                         $retval .= floor ($dayofyear / 7);
  758.                         break;
  759.                     case 'w':
  760.                         $retval .= $dayofweek;
  761.                         break;
  762.                     case 'X': // # of weekday for this month
  763.                         $retval .= floor (($day - 1) / 7) + 1;
  764.                         break;
  765.                     case 'x': // Total # of weekdays for this month
  766.                         $lastdayofweek = ((7 + $dayofweek + (($this->getDaysPerMonth ($year, $month) - $day) % 7)) % 7);
  767.                         if ((($dayofweek + 7) >= ($lastdayofweek - (($this->getDaysPerMonth ($year, $month) % 7) - 1)) + 7) && (($dayofweek + 7) <= ($lastdayofweek + 7))) {
  768.                             $retval .= 5;
  769.                         } else {
  770.                             $retval .= 4;
  771.                         }
  772.                         break;
  773.                     case 'Y':
  774.                         $retval .= $year;
  775.                         break;
  776.                     case 'y':
  777.                         $retval .= substr ($year, 2);
  778.                         break;
  779.                     case 'Z':
  780.                         $retval .= $timezone_info[2];
  781.                         break;
  782.                     case 'z':
  783.                         $retval .= $dayofyear;
  784.                         break;
  785.                     default:
  786.                         $retval .= substr ($format, $index, 1);
  787.                 }
  788.             }
  789.  
  790.             return $retval;
  791.         }

  795.          * Behaves exactly like PHP's implementation of unix getdate
  796.          * http://www.php.net/getdate
  797.          *
  798.          * @access public
  799.          * @param integer Timestamp since January 1st of beginYear in UTC

  802.         function getdate ($timestamp=0) {

  812.                 'hours'=>str_pad ($d[2], 2, '0', STR_PAD_LEFT), 'mday'=>str_pad ($d[3], 2, '0', STR_PAD_LEFT),

  814.         }

  816.         /**
