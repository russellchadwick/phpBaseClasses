  165.         /**
  166.          * Fetches some information about a form within a fetched page
  167.          *
  168.          * @access public
  169.          * @param string name of form to look for
  170.          * @return array (string method, string url, array data (input_name, input_value))
  171.          */
  172.         function getFormData ($form_name='') {
  173.             $form_data = array ();
  174.  
  175.             $keep_looking = 0;
  176.  
  177.             while (strpos ($this->_page, '<form ', $keep_looking) !== false) { 
  178.                 $form_begin = strpos ($this->_page, '<form ', $keep_looking);
  179.                 $form_end = strpos ($this->_page, '</form', $form_begin);
  180.                 $form_text = substr ($this->_page, $form_begin, ($form_end - $form_begin));
  181.  
  182.                 $form_tag = substr ($this->_page, $form_begin, (strpos ($this->_page, '>', $form_begin) - $form_begin));
  183.                 $form_properties = $this->getHTMLProperties ($form_tag);
  184.  
  185.                 if (($form_properties['name'] == $form_name) || (empty ($form_name))) {
  186.                     $form_data['method'] = $form_properties['method'];
  187.                     $form_data['url'] = $form_properties['action'];
  188.  
  189.                     if (substr ($form_data['url'], 0, 7) != 'http://') {
  190.                         $form_data['url'] = $this->_url . $form_data['url'];
  191.                     }
  192.  
  193.                     $begin = 0;
  194.                     while (strpos ($form_text, '<input', $begin) !== false) {
  195.                         $input_begin = strpos ($form_text, '<input', $begin);
  196.                         $input_end = strpos ($form_text, '>', $input_begin);
  197.  
  198.                         $input_tag = substr ($form_text, $input_begin, ($input_end - $input_begin));
  199.                         $input_properties = $this->getHTMLProperties ($input_tag);
  200.  
  201.                         $form_data['data'][$input_properties['name']] = $input_properties['value'];
  202.  
  203.                         $begin = $input_end;
  204.                     }
  205.  
  206.                     break;
  207.                 }
  208.  
  209.                 $keep_looking = $form_end;
  210.             }
  211.  
  212.             return $form_data;
  213.         }

  215.         /**
  216.          * Take in an html tag and returns its key->values in an array
  217.          *
  218.          * @access private
  219.          * @param string html tag
  220.          * @return array (key=>value)
  221.          */
  222.         function getHTMLProperties ($tag) {
  223.             $properties = array ();
  224.  
  225.             $tag_pieces = explode (' ', $tag);
  226.             array_shift ($tag_pieces);
  227.  
  228.             for ($index = 0; $index < count($tag_pieces); $index++) {
  229.                 if (substr_count ($tag_pieces[$index], '"') % 2 == 1) {
  230.                     $tag_pieces[$index] .= ' ' . $tag_pieces[$index+1];
  231.                     $tag_pieces[$index+1] = '';
  232.                 }
  233.             }
  234.  
  235.             for ($index = 0; $index < count($tag_pieces); $index++) {
  236.                 if (strpos($tag_pieces[$index], '=') !== false) {
  237.                     $name = substr ($tag_pieces[$index], 0, strpos ($tag_pieces[$index], '='));
  238.                     $value = str_replace ('&nbsp;', ' ', substr ($tag_pieces[$index], strpos($tag_pieces[$index], '=') + 1));
  239.  
  240.                     if ((substr ($value, 0, 1) == '"') && (substr ($value, -1) == '"')) {
  241.                         $value = substr ($value, 1, -1);
  242.                     }
  243.  
  244.                     $properties[$name] = $value;
  245.                 }
  246.             }
  247.  
  248.             if (!isset ($properties['name'])) {
  249.                 $properties['name'] = '';
  250.             }
  251.  
  252.             return $properties;
  253.         }

  255.         /**
  256.          * Submits a form based on form data from getFromData
  257.          *
  258.          * @access public
  259.          * @param array (string method, string url, array data (input_name, input_value)) from getFormData
  260.          * @param array (key=>value) items to override whats given from getFormData
  261.          * @return string data received from remote server
  262.          */
  263.         function submitForm ($form_data, $override_data) {
  264.             $post_data = '';
  265.  
  266.             $form_data['url'] .= '?';
  267.  
  268.             while (list ($name, $value) = each ($form_data['data'])) {
  269.                 if (isset ($override_data[$name])) {
  270.                     $value = $override_data[$name];
  271.                 }
  272.                 if (strtolower ($form_data['method']) == 'post') {
  273.                     $post_data .= $name . '=' . $value . '&';
  274.                 } else {
  275.                     $form_data['url'] .= $name . '=' . $value . '&';
  276.                 }
  277.             }
  278.  
  279.             if ((substr ($form_data['url'], -1) == '&') || (substr ($form_data['url'], -1) == '?')) {
  280.                 $form_data['url'] = substr ($form_data['url'], 0, -1);
  281.             }
  282.  
  283.             if (substr ($post_data, -1) == '&') {
  284.                 $post_data = substr ($post_data, 0, -1);
  285.             }
  286.  
  287.             return ($this->getURL ($form_data['url'], $post_data));
  288.         }

  290.         /**
  291.          * gets a page from a remote server
  292.          *
  293.          * @access public
  294.          * @param string path to remote server
  295.          * @param string list of post vars in normal get format var1=val1&var2=val2
  296.          * @param boolean Whether you would like the HTTP Headers included
  297.          * @return string data received from remote server
  298.          */
  299.         function getURL ($url, $postvars='', $headers=false) {
  300.             $this->_url = $url;
  301.             $relocate = '';
  302.  
  303.             $ch = curl_init();
  304.             curl_setopt ($ch, CURLOPT_URL, $this->_url);
  305.             curl_setopt ($ch, CURLOPT_REFERER, $this->_referrer);
  306.             curl_setopt ($ch, CURLOPT_HEADER, $headers);
  307.             curl_setopt ($ch, CURLOPT_RETURNTRANSFER, true);
  308.             curl_setopt ($ch, CURLOPT_USERAGENT, $this->_browser);
  309.             debug ('UrlFetch::getURL -- Cookie Data = ' . $this->getCookies (), 3);
  310.             curl_setopt ($ch, CURLOPT_COOKIE, $this->getCookies ());
  311.  
  312.             if (!empty ($postvars)) {
  313.                 debug ('UrlFetch::getURL -- Post Variables = ' . $postvars, 3);
  314.                 curl_setopt ($ch, CURLOPT_POST, 1);
  315.                 curl_setopt ($ch, CURLOPT_POSTFIELDS, $postvars);
  316.             }
  317.  
  318.             debug ('UrlFetch::getURL -- Calling: ' . $url, 3);
  319.             $this->_page = curl_exec ($ch);
  320.  
  321.             $cookie_count = 0;
  322.             foreach (explode ("\n", $this->_page) as $line) {
  323.                 if (substr ($line, 0, 12) == 'Set-Cookie: ') {
  324.                     $cookie_data = substr ($line, 12, strpos ($line, ';') - 12);
  325.                     $cookie_name = substr ($cookie_data, 0, strpos ($cookie_data, '='));
  326.                     $cookie_value = substr ($cookie_data, strpos ($cookie_data, '=') + 1);
  327.                     $this->_cookies[$cookie_name] = $cookie_value;
  328.                     $cookie_count++;
  329.                     debug ('UrlFetch::getURL -- Found Cookie: ' . $cookie_name, 3);
  330.                 }
  331.             }
  332.  
  333.             foreach (explode ("\n", $this->_page) as $line) {
  334.                 if (substr ($line, 0, 10) == 'Location: ') {
  335.                     $relocate = substr ($line, 10);
  336.                 }
  337.             }
  338.  
  339.             if (curl_error ($ch) > '') {
  340.                 debug ('UrlFetch::getURL -- Curl Error - ' . curl_error ($ch), 1);
  341.             }
  342.  
  343.             curl_close ($ch);
  344.  
  345.             $this->_referrer = $url;
  346.  
  347.             if (!empty ($relocate)) {
  348.                 debug ('UrlFetch::getURL -- Relocating to ' . $relocate, 3);
  349.                 $this->getURL ($relocate);
  350.             }
  351.  
  352.             return ($this->_page);
  353.         }
  354.     }
