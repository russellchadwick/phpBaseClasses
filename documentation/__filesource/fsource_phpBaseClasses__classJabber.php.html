  123.         function Jabber ($host, $username, $password, $blocking=false, $aim=false) {
  124.             $this->_host = $host;
  125.             $this->_username = $username;
  126.             $this->_password = $password;
  127.             $this->_blocking = $blocking;
  128.             $this->_aim = $aim;
  129.  
  130.             $this->_sock = fsockopen ($this->_host, '5222');
  131.  
  132.             if ($this->_sock) {
  133.                 stream_set_blocking ($this->_sock, true);
  134.                 stream_set_timeout ($this->_sock, 31536000);  // 365 days
  135.  
  136.                 $this->writeData ('<stream:stream to="' . $host . '" xmlns="jabber:client" xmlns:stream="http://etherx.jabber.org/streams" version="1.0" >');
  137.  
  138.                 sleep (1);
  139.  
  140.                 $stream_data = $this->readData ();
  141.                 $this->_stream_id = substr ($stream_data, (strpos ($stream_data, ' id=') + 5), ((strpos ($stream_data, ' xmlns=') - 1) - (strpos ($stream_data, ' id=') + 5)));
  142.  
  143.                 $this->login ();
  144.  
  145.                 stream_set_blocking ($this->_sock, $blocking);
  146.             } else {
  147.                 debug ('No Jabber Server!', 1);
  148.             }
  149.         }

  157.         function login () {
  158.             $this->writeData ('<iq id="jcl_11" type="get"><query xmlns="jabber:iq:auth"><username>' . $this->_username . '</username></query></iq>');

  165.  
  166.             $this->writeData ('<iq id="jcl_12" type="set"><query xmlns="jabber:iq:auth"><username>' . $this->_username . '</username><resource>phpBaseClasses classJabber.php</resource><digest>' . $password_hash . '</digest></query></iq>');
  167.  
  168.             sleep (2);

  170.             $this->readData ();

  179.             $this->readData ();

  192.             }

  195.         /**
