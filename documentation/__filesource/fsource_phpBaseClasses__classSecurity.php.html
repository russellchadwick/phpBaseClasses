  214.         function checkSessionVars () {

  223.                             $restricted = true;

  228.  
  229.                     if (($restricted) && ( (!isset ($_SESSION[$config['SESSION_REQUIRED_VAR']])) || ($_SESSION[$config['SESSION_REQUIRED_VAR']] == 0) || ($_SESSION[$config['SESSION_REQUIRED_VAR']] == '') )) {
  230.                         if (!empty ($config['SESSION_LOGIN_URI'])) {
  231.                             header ('Location: ' . $config['SESSION_LOGIN_URI'] . '?session_command=logout&message=' . urlencode ('Either you tried to access a restricted area without logging in, or you were inactive long enough your session was deleted.'));

  237.                 if ($session_var == 'last_access') {

  239.                         if ($restricted) {

  241.                                 if (!empty ($config['SESSION_LOGIN_URI'])) {
  242.                                     header ('Location: ' . $config['SESSION_LOGIN_URI'] . '?session_command=logout&message=' . urlencode ('Idle More Than ' . $config['SESSION_TIMEOUT'] . ' Minutes'));
  243.                                     die ();
  244.                                 }
  245.                             } else {
  246.                                 $_SESSION['last_access'] = mktime ();
  247.                             }
  248.                         } else {
  249.                             $_SESSION['last_access'] = mktime ();
  250.                         }
  251.                     } else {
  252.                         $_SESSION['last_access'] = mktime ();
  253.                     }
  254.                 }
  255.             }
