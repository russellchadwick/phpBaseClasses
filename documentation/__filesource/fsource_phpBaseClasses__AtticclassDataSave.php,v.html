  191.         function fixupData ($description, $column_name) {
  192.             global $db;
  193.  
  194.             $current_value = @$_POST[$column_name];
  195.  
  196.             switch ($description['data_type']) {
  197.                 case 'char':

  199.                     $current_value = $this->fixupText ($description, $current_value);
  200.                     if ($description['is_nullable'] == true) {
  201.                         $current_value = $this->fixupLength ($description, $current_value, 0, $description['data_length']);

  204.                     }                        
  205.                     if ($description['data_length'] > 300) {
  206.                         $current_value = $db->sqlEscape ($current_value, true);
  207.                     } else {

  210.                     break;
  211.                 case 'real':
  212.                 case 'money':
  213.                     if ((!empty($_POST[$column_name . '_whole'])) || (!empty($_POST[$column_name . '_decimal']))) {
  214.                         $current_value = $this->or0($_POST[$column_name . '_whole']) . '.' . $this->or0($_POST[$column_name . '_decimal']);
  215.                         if ($current_value == '.') $current_value = '';
  216.                     }
  217.                     break;
  218.                 case 'date':
  219.                     $current_value = $_POST[$column_name . '_year'] . '-' . $_POST[$column_name . '_month'] . '-' . $_POST[$column_name . '_day'];
  220.                     if ($current_value == '--') $current_value = '';

  222.                     break;
  223.                 case 'timestamp with time zone':
  224.                     $current_value = $_POST[$column_name . '_year'] . '-' . $_POST[$column_name . '_month'] . '-' . $_POST[$column_name . '_day'] . ' ' . $_POST[$column_name . '_hour'] . ':' . $_POST[$column_name . '_minute'] . ':' . $_POST[$column_name . '_second'] . ' ' . $_POST[$column_name . '_timezone'] . ':00';
  225.                     if ($current_value == '-- :: :00') $current_value = '';
  226.                     if ($_POST[$column_name . '_na'] == 'on') $current_value = '';
  227.                     break;
  228.             }
  229.             if (empty($_POST[$column_name])) {
  230.                 $_POST[$column_name] = $current_value;
  231.             }
  232.             $current_value = $this->fixupNull ($description, $column_name);
  233.  
  234.             return $current_value;
  235.         }
