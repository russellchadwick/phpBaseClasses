  307.                     if (!$DBCONNECTEDTOALL) {

  346.             }
  347.  
  348.             debug ('End Update: ' . $query, -1);

  382.                     $this->executeSql ('EXPLAIN ANALYZE ' . $query);
  383.                     $explain_count = pg_numrows ($this->_result);
  384.                     for ($x=0; $x < $explain_count; $x++) {
  385.                         $plan = pg_fetch_array ($this->_result, $x, PGSQL_ASSOC);

  387.                     }

  389.                     error_log ("\n", 3, $this->_dbExplainLog);
  390.                 }
  391.  
  392.                 if (!empty ($this->_result)) {
  393.                     pg_freeresult ($this->_result);
  394.                 }
  395.             } else if ($this->_dbType == 'MySQL') {
  396.                 if (!empty ($this->_result)) {
  397.                     $count = mysql_num_rows ($this->_result);
  398.                 } else {
  399.                     $count = 0;
  400.                 }
  401.  
  402.                 for ($x=1; $x <= $count; $x++) {
  403.                     $data[$x] = mysql_fetch_array ($this->_result, MYSQL_ASSOC);
  404.                 }
  405.  
  406.                 if (!empty ($this->_result)) {
  407.                     mysql_free_result ($this->_result);
  408.                 }
  409.             } else if ($this->_dbtype == 'Oracle') {
  410.                 if (!empty ($this->_result)) {
  411.                     $count = OCIFetchStatement ($this->_result, $data);
  412.                 } else {
  413.                     $count = 0;
  414.                 }
  415.  
  416.                 if (!empty ($this->_result)) {
  417.                     ocifreestatement ($this->_result);
  418.                 }
  419.             } else if ($this->_dbtype == 'Informix') {
  420.                 if (!empty ($this->_result)) {
  421.                     $count = 0;
  422.  
  423.                     $row = ifx_fetch_row ($this->_result, 'NEXT');
  424.                     while (is_array ($row)) {
  425.                         $count++;
  426.                         foreach ($row as $row_index => $row_data) {
