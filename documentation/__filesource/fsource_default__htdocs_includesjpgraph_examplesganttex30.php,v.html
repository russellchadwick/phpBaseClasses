  23. $graph->scale->week->SetStyle(WEEKSTYLE_FIRSTDAY);
  24.  
  25. // Change the scale font

  28.  
  29.  
  30. // Setup some data for the gantt bars
  31. $data = array(
  32.     array(0,"Group 1", "2001-10-29","2001-11-27",FF_FONT1,FS_BOLD,8),
  33.     array(1,"  Label 2", "2001-11-8","2001-12-14"),
  34.     array(2,"  Label 3", "2001-11-01","2001-11-8"),
  35.     array(4,"Group 2", "2001-11-07","2001-12-19",FF_FONT1,FS_BOLD,8),
  36.     array(5,"  Label 4", "2001-11-8","2001-12-19"),
  37.     array(6,"  Label 5", "2001-11-01","2001-11-8")
  38.     );
  39.  
  40. for($i=0; $i<count($data); ++$i) {
  41.     $bar = new GanttBar($data[$i][0],$data[$i][1],$data[$i][2],$data[$i][3],"[50%]",0.5);
  42.     if( count($data[$i])>4 )
  43.         $bar->title->SetFont($data[$i][4],$data[$i][5],$data[$i][6]);
  44.         
  45.     // If you like each bar can have a shadow

  47.     
  48.     // For illustration lets make each bar be red with yellow diagonal stripes

  50.     $bar->SetFillColor("red");
  51.     
  52.     // To indicate progress each bar can have a smaller bar within
  53.     // For illustrative purpose just set the progress to 50% for each bar
  54.     $bar->progress->Set(0.5);
  55.     
  56.     // Each bar may also have optional left and right plot marks

  58.     // of each bar
  59.     $bar->rightMark->SetType(MARK_FILLEDCIRCLE);
  60.     $bar->rightMark->SetFillColor("red");
  61.     $bar->rightMark->SetColor("red");
  62.     $bar->rightMark->SetWidth(10);
  63.     
  64.     // Title for the mark

  66.     $bar->rightMark->title->SetColor("white");

  68.     $bar->rightMark->Show();
  69.     
  70.     // ... and add the bar to the gantt chart
  71.     $graph->Add($bar);
